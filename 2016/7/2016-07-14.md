## Koa

`koa-compose`最后为什么返回`yield* next`

```js
function* gen(next) {
  console.log('gen')
  yield 1;
  yield next;
  console.log(this.a);
  yield 2;
}

function* gen2(next) {
  console.log('gen2')
  yield next;
  console.log('gen2 next')
}

let _self = {a: 1};
function* next() {
  let n = function* () {};
  
  n = gen.call(_self, n);
  n = gen2.call(_self, n);
  
  return yield* n;
}

let n = next();
let val;
console.log(n = n.next());
val = n.value;
console.log(val.next());
```
