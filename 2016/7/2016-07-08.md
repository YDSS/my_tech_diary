1. thunk函数
    - 有且只有一个参数是callback的函数
    - callback的第一个参数是error
    - thunkToPromise

        ```js
        function thunkToPromise(fn) {
          return new Promise((res, rej) => {
              fn((err, data) => {
                if (err) return rej(err);
                res(data);
              })
          })
        }
        
        function gen(a, fn) {
          return fn(null, a + 1);
        }
        
        function thunk(fn) {
          return gen(1, fn);
        }
        let tp = thunkToPromise(thunk);
        tp.then(ret => {
          console.log(ret);
        })
        ```
